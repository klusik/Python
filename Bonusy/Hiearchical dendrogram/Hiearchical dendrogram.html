# Scaling the data

# Linkage matrix
Z = linkage(scaled_data, method='ward')

# Plotting dendrogram
plt.figure(figsize=(10, 7))
dendrogram(Z)
plt.title("Hierarchical Clustering Dendrogram")
plt.xlabel('Configuration Index')
plt.ylabel('Distance')
plt.show()

# Cutting the dendrogram based on a chosen distance threshold
max_d = 25
clusters = fcluster(Z, max_d, criterion='distance')

# Apply PCA for visualization
pca = PCA(n_components=2)
principal_components = pca.fit_transform(scaled_data)

# Plotting the clusters in a 2D space based on PCA results
plt.figure(figsize=(10, 7))
scatter = plt.scatter(principal_components[:, 0], principal_components[:, 1], c=clusters, cmap='rainbow')

# Create a legend for the colors
handles, _ = scatter.legend_elements(prop="colors", alpha=0.6)
legend1 = plt.legend(handles, range(1, max(clusters)+1), loc="best", title="Clusters")

plt.title('PCA Cluster Visualization')
plt.xlabel('Principal Component 1')
plt.ylabel('Principal Component 2')
plt.show()
